Create TABLE USER_LOGIN(
 USER_ID VARCHAR2(10) NOT NULL,
 USER_PSWD VARCHAR2(10) NOT NULL,
 PRIMARY KEY(USER_ID));


CREATE TABLE OLNS_PROD_CATEGORIES(

 CATEGORY_ID NUMBER(10) NOT NULL,
 CATEGORY VARCHAR2(100) NOT NULL,
 PRIMARY KEY(CATEGORY_ID));


CREATE TABLE OLNS_PRODUCTS(
 PRODUCT_ID NUMBER(10),
 CATEGORY_ID NUMBER(10),
 NAME VARCHAR2(30),
 DESCRIPTION VARCHAR2(100),
 PRICE NUMBER(7,2),
 PRIMARY KEY(PRODUCT_ID),
FOREIGN KEY(CATEGORY_ID) REFERENCES OLNS_PROD_CATEGORIES(CATEGORY_ID) ON DELETE CASCADE);

CREATE TABLE OLNS_PROD_ORDER(
 ORDER_ID NUMBER(10),
 CONTACT_NAME VARCHAR2(100),
 DELIVERY_ADDRESS VARCHAR2(100),
 CC_NAME VARCHAR2(30),
 CC_NUMBER NUMBER(20),
 CC_EXPIRY_DATE DATE,
 PRIMARY KEY(ORDER_ID));
 
CREATE TABLE OLNS_PROD_ORDER_DETAILS(
 ORDER_DTL_ID NUMBER(10),
 ORDER_ID NUMBER(10),
 PRODUCT_ID NUMBER(10),
 ORDER_QTY NUMBER(10),
 ORDER_PRICE NUMBER(10),
 PRIMARY KEY(ORDER_DTL_ID, ORDER_ID));
